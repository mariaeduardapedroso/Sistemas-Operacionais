#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>
#include <sys/wait.h>

int main()
{

    int var = 0;
    int pid;

    pid = fork();

    // filho
    if (pid == 0)
    {
        var = 5;
        printf("Filho: %d", var);
    }
    else
    {
        if (pid > 0)
        {
            var += 10;
            printf("Pai: %d", var);
        }
        else
        {
            printf("Erro na criacao do filho");
        }
    }
}
